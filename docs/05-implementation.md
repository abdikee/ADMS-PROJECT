### 1. Calculate Total Marks for a Student

```sql
-- Calculate total marks obtained and total maximum marks for a student
SELECT 
  s.id AS student_id,
  s.first_name,
  s.last_name,
  s.admission_number,
  SUM(m.marks_obtained) AS total_marks_obtained,
  SUM(m.max_marks) AS total_max_marks,
  ROUND((SUM(m.marks_obtained) / SUM(m.max_marks) * 100), 2) AS percentage
FROM students s
JOIN marks m ON s.id = m.student_id
WHERE s.id = 1  -- Replace with actual student ID
GROUP BY s.id, s.first_name, s.last_name, s.admission_number;
```

### 2. Calculate Weighted Total Based on Exam Type

```sql
-- Calculate weighted total for a student based on exam type weightage
-- Midterm: 30%, Final: 50%, Quiz: 10%, Assignment: 10%
WITH student_marks AS (
  SELECT 
    m.student_id,
    m.subject_id,
    sub.name AS subject_name,
    sub.max_marks AS subject_max_marks,
    et.code AS exam_type_code,
    et.weightage,
    m.marks_obtained
  FROM marks m
  JOIN subjects sub ON m.subject_id = sub.id
  JOIN exam_types et ON m.exam_type_id = et.id
  WHERE m.student_id = 1  -- Replace with actual student ID
),
subject_weighted AS (
  SELECT 
    student_id,
    subject_id,
    subject_name,
    SUM(
      CASE 
        WHEN exam_type_code = 'MIDTERM' THEN (marks_obtained / subject_max_marks * 100) * 0.30
        WHEN exam_type_code = 'FINAL' THEN (marks_obtained / subject_max_marks * 100) * 0.50
        WHEN exam_type_code = 'QUIZ' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        WHEN exam_type_code = 'ASSIGNMENT' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        ELSE 0
      END
    ) AS weighted_percentage
  FROM student_marks
  GROUP BY student_id, subject_id, subject_name
)
SELECT 
  student_id,
  subject_name,
  ROUND(weighted_percentage, 2) AS weighted_percentage,
  CASE 
    WHEN weighted_percentage >= 90 THEN 'A'
    WHEN weighted_percentage >= 80 THEN 'B'
    WHEN weighted_percentage >= 60 THEN 'C'
    WHEN weighted_percentage >= 50 THEN 'D'
    ELSE 'F'
  END AS grade,
  CASE 
    WHEN weighted_percentage >= 50 THEN 'PASS'
    ELSE 'FAIL'
  END AS status
FROM subject_weighted
ORDER BY subject_name;
```

### 3. Calculate Average Percentage Across All Subjects

```sql
-- Calculate overall average percentage for a student
WITH student_marks AS (
  SELECT 
    m.student_id,
    m.subject_id,
    sub.max_marks AS subject_max_marks,
    et.code AS exam_type_code,
    m.marks_obtained
  FROM marks m
  JOIN subjects sub ON m.subject_id = sub.id
  JOIN exam_types et ON m.exam_type_id = et.id
  WHERE m.student_id = 1  -- Replace with actual student ID
),
subject_weighted AS (
  SELECT 
    student_id,
    subject_id,
    SUM(
      CASE 
        WHEN exam_type_code = 'MIDTERM' THEN (marks_obtained / subject_max_marks * 100) * 0.30
        WHEN exam_type_code = 'FINAL' THEN (marks_obtained / subject_max_marks * 100) * 0.50
        WHEN exam_type_code = 'QUIZ' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        WHEN exam_type_code = 'ASSIGNMENT' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        ELSE 0
      END
    ) AS weighted_percentage
  FROM student_marks
  GROUP BY student_id, subject_id
)
SELECT 
  s.id AS student_id,
  s.first_name,
  s.last_name,
  s.admission_number,
  ROUND(AVG(sw.weighted_percentage), 2) AS average_percentage,
  CASE 
    WHEN AVG(sw.weighted_percentage) >= 90 THEN 'A'
    WHEN AVG(sw.weighted_percentage) >= 80 THEN 'B'
    WHEN AVG(sw.weighted_percentage) >= 60 THEN 'C'
    WHEN AVG(sw.weighted_percentage) >= 50 THEN 'D'
    ELSE 'F'
  END AS overall_grade,
  CASE 
    WHEN AVG(sw.weighted_percentage) >= 50 THEN 'PASS'
    ELSE 'FAIL'
  END AS status
FROM students s
JOIN subject_weighted sw ON s.id = sw.student_id
WHERE s.id = 1  -- Replace with actual student ID
GROUP BY s.id, s.first_name, s.last_name, s.admission_number;
```

        WHEN exam_type_code = 'QUIZ' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        WHEN exam_type_code = 'ASSIGNMENT' THEN (marks_obtained / subject_max_marks * 100) * 0.10
        ELSE 0
      END
    ) AS weighted_percentage
  FROM student_marks
  GROUP BY student_id, class_id, subject_id
),
student_averages AS (
  SELECT 
    student_id,
    class_id,
    AVG(weighted_percentage) AS average_percentage
  FROM subject_weighted
  GROUP BY student_id, class_id
)
SELECT 
  s.id AS student_id,
  s.first_name,
  s.last_name,
  s.admission_number,
  c.name AS class_name,
  ROUND(sa.average_percentage, 2) AS average_percentage,
  RANK() OVER (PARTITION BY sa.class_id ORDER BY sa.average_percentage DESC) AS rank,
  CASE 
    WHEN sa.average_percentage >= 50 THEN 'PASS'
    ELSE 'FAIL'
  END AS status
FROM students s
JOIN student_averages sa ON s.id = sa.student_id
JOIN classes c ON sa.class_id = c.id

    et.code AS exam_type_code,

        WHEN exam_type_code = 'QUIZ' THEN (marks_obtained / subject_max_marks * 100) * 0.10

      END
    ) AS weighted_percentage

),

      WHEN AVG(weighted_percentage) >= 60 THEN 'C'
      WHEN AVG(weighted_percentage) >= 50 THEN 'D'
      ELSE 'F'
    END AS grade
  FROM subject_weighted
  GROUP BY student_id
)

  CASE 
    WHEN grade = 'A' THEN 'Excellent (90-100%)'
    WHEN grade = 'B' THEN 'Very Good (80-89.99%)'
    WHEN grade = 'C' THEN 'Good (60-79.99%)'

  ROUND(COUNT(*)::numeric / (SELECT COUNT(*) FROM student_averages)::numeric * 100, 2) AS percentage
FROM student_averages
GROUP BY grade
ORDER BY 
  CASE grade
    WHEN 'A' THEN 1
    WHEN 'B' THEN 2
    WHEN 'C' THEN 3
    WHEN 'D' THEN 4
    WHEN 'F' THEN 5
  END;
